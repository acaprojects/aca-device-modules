def down(index = 1)
    if self[:"screen#{index}"] == :down
        down_only(index)
    else
        stop(index)
        down_only(index)
    end
end

def down_only(index = 1)
    do_send({
        state: :down,
        body: {"Down#{index}" => "Down"},
        name: :"position#{index}",

    })
