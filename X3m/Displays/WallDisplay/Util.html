def dig(hash, key, *keys)
    value = hash[key]
    if value.nil? || keys.empty?
        value
    else
        dig value, *keys
    end
end

def encode(value, length:)
    value
        .to_s(16)
        .upcase
        .rjust(length, '0')
        .bytes
end
