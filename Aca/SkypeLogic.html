def accept_call
    self[:accept_call] += 1

    if not self[:in_call]
        schedule.in('1s') do
            mute(true)
        end
    end
end

def mute(state)
    self[:mute] = !!state

    if @mic_mutes && !@mic_mutes.empty?
        system[:Mixer].mute(@mic_mutes, self[:mute])
    end
end

def on_load
    self[:accept_call] = 0
