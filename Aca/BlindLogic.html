def on_update
    blinds = setting :blinds

    hashable_settings = blinds.each_with_index.map do |blind, idx|
        name = blind[:title] || "Blind #{idx + 1}"
        [name.to_sym, blind.slice(:module, :up, :stop, :down)]
    end

    self[:controls] = Hash[hashable_settings]
end
