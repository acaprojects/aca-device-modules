def self.task
    reactor = ::Libuv::Reactor.current
    d = reactor.defer
    THREAD_POOL.post do
        begin
            d.resolve(yield)
        rescue Exception => e
            d.reject(e)
        end
    end
    promise = d.promise
    promise
end
