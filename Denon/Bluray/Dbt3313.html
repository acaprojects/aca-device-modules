def power(state, options = {})
	state = is_affirmative?(state)
	options.merge!({
		delay: 5500   # Delay at least 5 seconds according to manual
	})
	if state == On
		do_send(COMMANDS[:power_on], options)
	else
		do_send(COMMANDS[:power_off], options)
	end
end
