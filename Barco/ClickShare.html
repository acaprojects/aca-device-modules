def audio_enabled(val = nil)
    value = is_affirmative?(val)
    do_send '/v1.0/Audio/Enabled', value: value do |resp|
        if resp
            self[:audio_enabled] = resp[:data][:value]
        else
            self[:audio_enabled] = value
        end
    end
end
