def media_source(mode)
    the_mode = modes.to_s
    if the_mode == 'live'
        do_send 'live'
    elsif Modes.include? the_mode
        do_send 'local', the_mode
    else
        raise "invalid mode #{the_mode}"
    end
end

def mute(state = true)

        do_send 'txdisable'
    else

    end
end
