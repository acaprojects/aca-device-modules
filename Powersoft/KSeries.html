def mute(index, value = true)
    val = is_affirmative?(value) ? 0x31 : 0x30
    do_send(0x6d, 0x30 + index.to_i, val)
end

def power(state)
    target = is_affirmative?(state)
    promise = if target
        do_send(0x70, 0x31)
    else
        do_send(0x70, 0x30)
    end
    promise.then do
        self[:power] = target
    end
end
